let cardsData=JSON.parse(localStorage.getItem("cmcards"))||[];

function play(){tap.currentTime=0;tap.play()}

function render(){
cards.innerHTML="";
cardsData.forEach((c,i)=>{
const d=document.createElement("div");
d.className="card";
d.innerHTML=`
<div class="card-header">
<div>Banco Metropolitano de Cuba</div>
<div>üÖ±Ô∏èüÖ≤</div>
</div>
<div class="card-number">${c.number}</div>
<div class="card-footer">
<div>${c.name}</div>
<div>${c.expiry}</div>
</div>`;
d.onclick=()=>{
play();
if(confirm("¬øEliminar tarjeta?")){
cardsData.splice(i,1);
save();
}
}
cards.appendChild(d);
});
}

function save(){
localStorage.setItem("cmcards",JSON.stringify(cardsData));
render();
}

function saveCard(){
cardsData.push({
name:name.value||"Usuario",
number:number.value,
expiry:expiry.value
});
play();
save();
closeModal();
}

function toggleMode(){
document.body.classList.toggle("dark");
modeBtn.textContent=document.body.classList.contains("dark")?"‚òÄÔ∏è":"üåô";
play();
}

function openModal(){modal.style.display="flex"}
function closeModal(){modal.style.display="none"}
function openSettings(){settings.style.display="flex"}
function closeSettings(){settings.style.display="none"}

render();
</script>
